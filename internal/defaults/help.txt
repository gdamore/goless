[1;4mGeneral[0m
  h F1            help
  q Q Esc         quit
  r ^L            repaint

[1;4mNavigation[0m
  Enter e ^E ^J ^N
  ↑ ↓ j k         move one line forward/backward
  y ^Y ^K ^P      move one line backward
  PgDn ^F ^V
  f Space         forward one page
  PgUp ^B Alt-V b w backward one page
  Ctrl-D d        forward half page
  Ctrl-U u        backward half page
  Home 0          jump to line start
  End $           jump to line end
  g <             jump to top
  G >             jump to bottom
  ← → l           scroll horizontally faster
  Shift-← Shift-→ scroll horizontally by 1
  W               toggle wrap
  F               follow the end of the document

[1;4mSearch[0m
  /               search forward
  ?               search backward
  n N             next/previous match
  F2              cycle case mode (smart/case/nocase)
  F3              cycle match mode (substring/word/regex)

[1;4mPrompt Editing[0m
  ← →             move within the prompt
  Ctrl-B Ctrl-F   move backward/forward one character
  Home End        jump to start/end of prompt input
  Ctrl-A Ctrl-E   jump to start/end of prompt input
  Backspace Del   delete left/right
  Ctrl-K          delete to end of prompt input
  Ctrl-W          delete backward one word
  Ins             toggle insert/overwrite mode
  ↑ ↓             recall prompt history
  Ctrl-U          delete to beginning of prompt input

  :set wrap               soft-wrap long lines
  :set list               show hidden-character markers
  :set squeeze            collapse adjacent blank lines
  :set tabstop=[3mnum[0m        set tab width
  :set pinlines=[3mnum[0m       pin top lines
  :set pincols=[3mnum[0m        pin left columns
  :set ignorecase         search without case sensitivity
  :set smartcase          search smart-case
  :set searchmode=[3mmode[0m    set match mode (sub|word|regex)
